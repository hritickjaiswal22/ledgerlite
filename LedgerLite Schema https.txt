v2

currencies {
 id: UUID (primary-key, unique, not-null)
 name: VARCHAR(256) (unique, not-null)
 code: VARCHAR(3) (unique, not-null)
 symbol: VARCHAR(5) (unique, not-null)
 created_at: TIMESTAMPTZ (default now())
}

users {
 id: UUID (primary-key, unique,not-null)
 email: VARCHAR(256) (unique,not-null)
 password: TEXT (not-null,encrypted password using bcrypt)
 created_at: TIMESTAMPTZ (default now())
 updated_at: TIMESTAMPTZ (default now())
 selected_currency_id: UUID (not-null,foreign-key, what currency should be used for an account will be taken as input during signup)
}

enum ACCOUNT_TYPE {
  cash
  bank
  credit_card
}

accounts {
 id: UUID (primary-key, unique,not-null)
 user_id: UUID (foreign-key,not-null)
 name: VARCHAR(256,not-null)
 account_type: ACCOUNT_TYPE (not-null)
 balance: NUMERIC(15, 2) (Default 0.00) (check that for cash and bank >= 0)
 created_at: TIMESTAMPTZ (default now())
 updated_at: TIMESTAMPTZ (default now())
 
 [index should be created upon (user_id) and covering index on (name,balance)]
 [(user_id,name) should be unique in a table so that users cannot create duplicate accounts]
}

categories {
 id: UUID (primary-key, unique,not-null)
 user_id: UUID (foreign-key,not-null)
 name: VARCHAR(256) (,not-null)
 created_at: TIMESTAMPTZ (default now())
 updated_at: TIMESTAMPTZ (default now())
 
 [index should be created upon (user_id) and covering index on (name)]
 [(user_id,name) should be unique in a table so that users cannot create duplicate categories]
}

enum TRANSACTION_TYPE {
  income
  expense
}

transactions {
 id: UUID (primary-key, unique,not-null)
 account_id: UUID (foreign-key,not-null)
 user_id: UUID (foreign-key,not-null)
 amount: NUMERIC(15, 2) (>= 0 db level check,not-null)
 category_id: UUID (foreign-key,not-null)
 transaction_type: TRANSACTION_TYPE (not-null)
 transaction_date: TIMESTAMPTZ (not-null)
 description: VARCHAR(512)
 created_at: TIMESTAMPTZ (default now())
 
 [index on (user_id,transaction_date,category_id) INCLUDE (amount) as covering index where transaction_type=TRANSACTION_TYPE.income as partial index]
 [index on (user_id,transaction_date,category_id) INCLUDE (amount) as covering index where transaction_type=TRANSACTION_TYPE.expense as partial index]
}

budgets {
 id: UUID (primary-key, unique,(not-null))
 category_id: UUID (foreign-key,(not-null))
 user_id: UUID (foreign-key,(not-null))
 amount: NUMERIC(15, 2) (>= 0 db level check,(not-null))
 month: SMALLINT (1 to 12,(not-null))
 year: SMALLINT (not-null)
 created_at: TIMESTAMPTZ (default now())
 updated_at: TIMESTAMPTZ (default now())
 
 // [index should be created upon (user_id,month,year) including (category_id,amount) as covering index]
 [(user_id,category_id,month,year) should be unique in a table so that users cannot create duplicate budgets]
}
